<!DOCTYPE html>
<html>
<body>
  <h1>Hello Stylizer!</h1>
  
  <jsg-stylizer 
    is-development="true"
    default-primary-font="Roboto">
  </jsg-stylizer>
  
  <script type="module">
    import '@jsg/stylizer';
    
    document.querySelector('jsg-stylizer')
      .addEventListener('font-changed', (e) => {
        console.log('Font changed:', e.detail);
      });
  </script>
</body>
</html>
import '@jsg/stylizer';

function App() {
  const handleFontChange = (e) => {
    console.log('Font changed:', e.detail);
  };
  
  return (
    <div>
      <h1>My React App</h1>
      <jsg-stylizer
        is-development={import.meta.env.DEV}
        default-primary-font="Roboto"
        onFont-changed={handleFontChange}
      />
    </div>
  );
}
